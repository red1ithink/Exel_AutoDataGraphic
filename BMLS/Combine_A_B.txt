Sub CombineColumnsAandB()

    ' 변수 선언
    Dim ws As Worksheet
    Dim lastRow As Long
    Dim i As Long
    
    ' 현재 활성화된 시트를 대상으로 설정
    Set ws = ThisWorkbook.ActiveSheet
    
    lastRow = ws.Cells(ws.Rows.Count, "A").End(xlUp).Row
    
    ' 1행부터 마지막 행까지 반복
    ' 만약 1행이 제목(Header)이라서 2행부터 시작해야 한다면 For i = 2 To lastRow 로 변경하세요.
    For i = 1 To lastRow
        
        ' A열과 B열의 셀이 모두 비어있지 않은 경우에만 실행
        If Not IsEmpty(ws.Cells(i, "A").Value) And Not IsEmpty(ws.Cells(i, "B").Value) Then
        
            ' C열에 A열 값 & "-" & B열 값을 합쳐서 입력
            ws.Cells(i, "C").Value = ws.Cells(i, "A").Value & "-" & ws.Cells(i, "B").Value
            
        End If
        
    Next i
    
    MsgBox "A열과 B열의 데이터 조합이 완료되었습니다.", vbInformation

End Sub
